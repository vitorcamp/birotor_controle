#include "mbed.h"
#include "Servo.h"

InterruptIn button(USER_BUTTON);
DigitalOut degrau(PA_10);

bool _btnFunction=0;
int counter = 0;

void pressed()
{
   _btnFunction = !_btnFunction;
}
 
int main() {
     Servo motor1(PB_10);
     Servo motor2(PB_4);
     button.fall(&pressed);
     motor1.write(0);
     motor2.write(0);
     wait_ms(10000);
     motor1.write(0.1);
     motor2.write(0.1);
     degrau = 0;
     while(1) 
     {
        if(_btnFunction)
        {
            
            motor1.write(0.2);
            degrau =1;
        }
        else
        {
            degrau = 0;
            motor1.write(0.1);    
        }
            
     }
}
